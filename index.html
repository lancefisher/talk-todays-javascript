// in lib1.js
var lib1 = {};
lib1.log = function(s) {
  console.log(s);
}

var lib2 = {};

}

"Real" Modules

  • AMD
  • CJS

AMD


// define provided by a loader (require.js)
define(function () {
  var prefix = 'log: ';
  var lib1 = {
    name: 'lib1',
    log: function(s) {
      console.log(s);
    }
  }

  return lib1;
});

AMD


define(function (require) {
  var lib1 = require('lib1');
  console.log('lib1.name', lib1.name);
  var lib2 = {
    log: function(s, a) {
      console.log(s + a);
    }
  }

  return lib2;
});

AMD Notes

  • No build necessary (but you should build)
  • Doesn't pollute the global namespace
  • Easy to debug

CommonJS

(or node-style) 


var prefix = 'log: ';
function log(s){
  console.log(prefix + s);
}

// exports provided by module loader
exports.name = 'lib1';
exports.log = log;

CJS


// require provided by module loader
var lib1 = require('./lib1.js');
console.log('lib1.name', lib1.name);

function log(s){
  console.log(prefix + s);
}

exports.log = log;

CJS Notes

  • Build Necessary
  • Simple
  • Familiar to node developers
  • No config
